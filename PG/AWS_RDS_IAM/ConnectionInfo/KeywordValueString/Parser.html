def parse
  result = {}

  skip_whitespace
  until @buffer.eos?
    key = parse_key
    skip_whitespace
    assert_followed_by_equals_sign key
    skip_whitespace
    value = parse_value
    skip_whitespace

    result[key] = value
  end

  @buffer.reset
  result
end
